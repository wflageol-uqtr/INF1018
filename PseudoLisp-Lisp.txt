Exemple 1

(defun double (x)
	(+ x x))

(defun quadruple (x)
	(double (double x)))

(let d2 (double 2))
(let d4 (double 4))
(let q6 (quadruple 6))

(printfn d2)
(printfn d4)
(printfn q6)

(defun fibo (n)
  (if (<= n 2)
      1
      (+ (fibo (- n 2))
         (fibo (- n 1)))))

(fibo 6)
