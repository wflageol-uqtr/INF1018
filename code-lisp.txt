Exemple 1

(defun triple (x)
	(+ x (+ x x)))

(defun sextuple (x)
	(triple (triple x)))

(let d2 (triple 2))
(let d4 (triple 4))
(let q6 (sextuple 6))

(printfn d2)
(printfn d4)
(printfn q6)

Exemple Fonctions Impures

(let value 0)

(defun change-value (v)
  (set value v))
  
(defun indirect-change ()
  (change-value 12))
  
(defun map-change ()
  (map change-value (list 1 2 3)))
